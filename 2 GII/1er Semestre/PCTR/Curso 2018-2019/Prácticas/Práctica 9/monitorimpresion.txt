monitor impresoras;
  var                         
    i, impresoras: integer;
    libre: array[1..3] of boolean;
    imp: condition;
    
  procedure integer take_print();
  begin
    if(impresoras==0) then wait(imp); 
    n:=1;
    while not(libre[n]) do n++;
    libre[n]:=false;
    impresoras--;
    return n;
  end;
  
  procedure drop_print(var n: integer);
  begin
    libre[n]:=true;
    impresoras++;
    send(imp);         
  end;
  
  begin
    for i:=1 to 3 do
      libre[i]:=true;
    impresoras:=3;  
  end.
